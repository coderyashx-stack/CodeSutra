Functions:

    return a + b;
  }
  print(add(3, 4));

  for (i in range(5)) {
    print(i);  # 0, 1, 2, 3, 4
  }

Conditionals:
  if (x > 10) {
    print("big");
  } else {
    print("small");
  }

  arr = [1, 2, 3];
  length(arr);     # 3
  push(arr, 4);    # [1, 2, 3, 4]
  reverse(arr);    # [4, 3, 2, 1]

  text = "hello";
  upper(text);     # "HELLO"
  length(text);    # 5
  contains(text, "ell");  # true

  person = {name: "Bob", age: 30};
  person.name;     # "Bob"
  keys(person);    # ["name", "age"]
